class Operation {
/**
 * Represents an operation.
 * @constructor
 * @param {number} x - The title of the book.
 * @param {number} y - The author of the book.
 */
  constructor(x, y) {
    this.x = x;
    this.y = y;
  }

  /**
   * Returns sum of x and y
   * @method
   */
  sum() {
    return this.x + this.y;
  }
}
module.exports = Operation;
